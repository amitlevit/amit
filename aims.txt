<script>
const canvas = document.getElementById('canvas');
const ctx    = canvas.getContext('2d');
let W = 0, H = 0, MAP_W = 2280, MAP_H = 1190;
let gameState = null, wsConnected = false;

function resize() {
    const area = document.getElementById('game-area');
    canvas.width  = W = area.clientWidth;
    canvas.height = H = area.clientHeight;
    if (gameState) render();
}
new ResizeObserver(resize).observe(document.getElementById('game-area'));
resize();

const sx = (x) => x * W / MAP_W;
const sy = (y) => y * H / MAP_H;

const WS = location.protocol === 'https:' ? 'wss:' : 'ws:';
let ws;
function connect() {
    ws = new WebSocket(`${WS}//${location.host}`);
    ws.onopen  = () => { wsConnected = true; setDot('on','CONNECTED'); };
    ws.onmessage = (e) => { try { handle(JSON.parse(e.data)); } catch {} };
    ws.onclose = () => { wsConnected = false; setDot('','RECONNECTING…'); setTimeout(connect,2000); };
    ws.onerror = () => ws.close();
}
connect();

function handle(msg) {
    if (msg.type === 'state') {
        if (msg.map) { MAP_W = msg.map.width; MAP_H = msg.map.height; }
        gameState = msg; updateHUD(); render();
    } else if (msg.type === 'auth_result') showToast(msg.ok?`Auth: ${msg.name}`:`Failed: ${msg.message}`);
    else if (msg.type === 'admin_event') {
        if (msg.action === 'music') { toggleMusic(); setFeedback('♪'); return; }
        showToast(`Admin: ${msg.action.toUpperCase()}`); setFeedback(`✓ ${msg.action}`);
        if (msg.action === 'start' && _pendingAdminStart) { _pendingAdminStart = false; startRec(); }
        if (msg.action === 'reset') stopRec();
    }
    else if (msg.type === 'game_over') { showToast(`🏆 WINNER: ${msg.winner?.toUpperCase()}`); stopRec(); }
    else if (msg.type === 'error') { showToast(msg.message); setFeedback(`✗ ${msg.message}`); }
}

function updateHUD() {
    if (!gameState) return;
    const { running, paused, tickId, scores, gameOver, winner } = gameState;
    document.getElementById('ftick').textContent = `TICK: ${tickId}`;
    document.getElementById('sred').textContent  = scores?.red  || 0;
    document.getElementById('sblue').textContent = scores?.blue || 0;
    let label = 'IDLE';
    if (gameOver)         { label = `WINNER: ${winner?.toUpperCase()}`; setDot('on','GAME OVER'); }
    else if (running && !paused) { label = 'RUNNING'; setDot('run','RUNNING'); }
    else if (paused)      { label = 'PAUSED';  setDot('on','PAUSED'); }
    else                  { setDot('on','CONNECTED'); }
    document.getElementById('fstat').textContent = `STATUS: ${label}`;

    const ents = gameState.entities || [];
    const html  = ents.map(e =>
        `<div class="sr"><span class="sn" style="color:${e.team==='red'?'#dc2626':'#2563eb'}">${e.name}</span><span class="sv">${e.type==='goal'?'TARGET':e.type.toUpperCase()}</span></div>`
    ).join('');
    document.getElementById('sl').innerHTML = html || '—';
    buildRoleTable(gameState.roster);
}

function render() {
    ctx.clearRect(0, 0, W, H);
    ctx.fillStyle = '#060608';
    ctx.fillRect(0, 0, W, H);

    // Centre dividing line
    ctx.strokeStyle = '#1a1a2e';
    ctx.lineWidth = 1;
    ctx.setLineDash([6,4]);
    ctx.beginPath(); ctx.moveTo(W/2, 0); ctx.lineTo(W/2, H); ctx.stroke();
    ctx.setLineDash([]);

    // Team half tints
    ctx.fillStyle = 'rgba(220,38,38,0.04)';
    ctx.fillRect(0, 0, W/2, H);
    ctx.fillStyle = 'rgba(37,99,235,0.04)';
    ctx.fillRect(W/2, 0, W/2, H);

    if (!gameState || !gameState.entities) return;

    const cfg = gameState.config || { goalRadius:22, aimRadius:16, hitDistance:22 };

    for (const e of gameState.entities) {
        const ex = sx(e.pos.x), ey = sy(e.pos.y);
        const r  = e.type === 'goal' ? Math.max(8, sx(cfg.goalRadius)) : Math.max(6, sx(cfg.aimRadius));
        const col = e.team === 'red' ? '#dc2626' : '#2563eb';

        ctx.shadowColor = col; ctx.shadowBlur = 12;
        ctx.fillStyle = col;
        ctx.beginPath();
        if (e.type === 'goal') {
            ctx.arc(ex, ey, r, 0, Math.PI*2);
        } else {
            // Diamond for aim
            ctx.moveTo(ex, ey - r);
            ctx.lineTo(ex + r, ey);
            ctx.lineTo(ex, ey + r);
            ctx.lineTo(ex - r, ey);
            ctx.closePath();
        }
        ctx.fill();
        ctx.shadowBlur = 0;

        // Velocity arrow
        const spd = Math.hypot(e.vel.x, e.vel.y);
        if (spd > 5) {
            const scale = Math.min(40, spd * 0.08);
            const vx = e.vel.x / spd * scale, vy = e.vel.y / spd * scale;
            ctx.strokeStyle = col; ctx.lineWidth = 1.5; ctx.globalAlpha = 0.55;
            ctx.beginPath(); ctx.moveTo(ex, ey); ctx.lineTo(ex + sx(vx)*10, ey + sy(vy)*10); ctx.stroke();
            ctx.globalAlpha = 1;
        }

        // Label
        ctx.fillStyle = '#fff';
        ctx.font = `bold ${Math.max(10,Math.floor(r*0.85))}px Courier New`;
        ctx.textAlign = 'center'; ctx.textBaseline = 'middle';
        ctx.fillText(e.name[0], ex, ey);
        // Name below entity (all types)
        ctx.fillStyle = col;
        ctx.font = `bold ${Math.max(14, Math.floor(r * 1.1))}px Courier New`;
        ctx.textBaseline = 'top';
        ctx.fillText(e.name, ex, ey + r + 3);
    }  // end for
}  // end render

function adminAct(action) {
    if (!wsConnected) { showToast('Not connected'); return; }
    if (action === 'start') _pendingAdminStart = true;
    ws.send(JSON.stringify({type:'admin',password:document.getElementById('ap').value,action}));
}
function setFeedback(t) { const el=document.getElementById('afb'); el.textContent=t; setTimeout(()=>el.textContent='',3000); }
function dismissSplash() {
    document.getElementById('spl').style.display='none';
}
function toggleMusic() {
    const bgm=document.getElementById('bgm');
    bgm.paused?(bgm.currentTime=0,bgm.play().then(()=>updateMusicBtn(true)).catch(()=>{})):(bgm.pause(),updateMusicBtn(false));
}
function updateMusicBtn(on) { const b=document.getElementById('mbtn'); b.textContent=on?'♪ ON':'♪ OFF'; b.classList.toggle('play',on); }
function setDot(cls,txt) { document.getElementById('dot').className=cls; document.getElementById('stxt').textContent=txt; }
let _tt=null;
function showToast(t) { const el=document.getElementById('toast'); el.textContent=t; el.classList.add('show'); clearTimeout(_tt); _tt=setTimeout(()=>el.classList.remove('show'),2800); }

// ── Role assignment table ──────────────────────────────────────────────────────
const ROLE_LABELS = {
    'red-aim':  '🔴 CHASER',
    'red-goal': '🔴 TARGET',
    'blue-aim': '🔵 CHASER',
    'blue-goal':'🔵 TARGET',
    'out':      '— OUT',
};
function buildRoleTable(roster) {
    const tbody = document.getElementById('role-body');
    if (!tbody || !roster) return;
    tbody.innerHTML = roster.map(({ name, role }) => `
        <tr>
          <td>${name}</td>
          <td><select class="role-select" data-name="${name}">
            ${Object.entries(ROLE_LABELS).map(([v,l]) =>
                `<option value="${v}"${role===v?' selected':''}>${l}</option>`
            ).join('')}
          </select></td>
        </tr>`).join('');
}
function applyRoles() {
    if (!wsConnected) { showToast('Not connected'); return; }
    const assignments = {};
    document.querySelectorAll('.role-select').forEach(s => { assignments[s.dataset.name] = s.value; });
    ws.send(JSON.stringify({ type:'admin', password:document.getElementById('ap').value, action:'assign', assignments }));
}

// ── Video Recording ───────────────────────────────────────────────────────────
let _pendingAdminStart = false, _recorder = null, _recChunks = [];
function startRec() {
    if (_recorder && _recorder.state !== 'inactive') return;
    _recChunks = [];
    try {
        const stream = canvas.captureStream(30);
        const mime = ['video/webm;codecs=vp9','video/webm;codecs=vp8','video/webm']
            .find(m => MediaRecorder.isTypeSupported(m)) || 'video/webm';
        _recorder = new MediaRecorder(stream, {mimeType: mime});
        _recorder.ondataavailable = e => { if (e.data.size > 0) _recChunks.push(e.data); };
        _recorder.onstop = () => {
            const blob = new Blob(_recChunks, {type: _recorder.mimeType});
            const name = `AIM_${new Date().toISOString().replace(/[:.]/g,'-')}.webm`;
            const a = Object.assign(document.createElement('a'), {href: URL.createObjectURL(blob), download: name});
            a.click(); setTimeout(() => URL.revokeObjectURL(a.href), 5000);
            showToast('📹 Recording saved');
        };
        _recorder.start(1000);
        showToast('🔴 Recording started');
    } catch(e) { showToast('⚠ Recording not supported'); }
}
function stopRec() {
    if (_recorder && _recorder.state === 'recording') { _recorder.stop(); }
}

window.auth  = (code) => ws?.readyState===1 && ws.send(JSON.stringify({type:'auth',code}));
window.force = (entityId,x,y) => ws?.readyState===1 && ws.send(JSON.stringify({type:'cmd',entityId,force:{x,y}}));
</script>
