<script>
// ── Canvas ────────────────────────────────────────────────────────────────────
const canvas = document.getElementById('canvas');
const ctx    = canvas.getContext('2d');
let W = 0, H = 0, GRID_W = 38, GRID_H = 25;

// ── State ────────────────────────────────────────────────────────────────────
let gameState   = null;
let wsConnected = false;

function resize() {
    const board = document.getElementById('board');
    const scW   = 240; // scoreboard width + margin
    W = Math.max(100, board.clientWidth  - scW);
    H = Math.max(100, board.clientHeight - 4);
    canvas.width  = W;
    canvas.height = H;
    if (gameState) render();
}
new ResizeObserver(resize).observe(document.getElementById('board'));
resize();

// Player colors (22 distinct)
const COLORS = [
    '#00f2fe','#ff6b6b','#ffd93d','#6bcb77','#4d96ff','#ff6b9d',
    '#c77dff','#f77f00','#aef359','#fb5607','#8ecae6','#ffb703',
    '#06d6a0','#ef476f','#118ab2','#ffd166','#e9c46a','#264653',
    '#2a9d8f','#e76f51','#a8dadc','#457b9d',
];

// Pre-build color map for player names (filled after first state)
const playerColor = {};

// ── WebSocket ─────────────────────────────────────────────────────────────────
const WS = location.protocol === 'https:' ? 'wss:' : 'ws:';
let ws;
function connect() {
    ws = new WebSocket(`${WS}//${location.host}`);
    ws.onopen  = () => { wsConnected = true; setDot('on', 'CONNECTED'); };
    ws.onmessage = (e) => { try { handle(JSON.parse(e.data)); } catch {} };
    ws.onclose = () => { wsConnected = false; setDot('', 'RECONNECTING…'); setTimeout(connect, 2000); };
    ws.onerror = () => ws.close();
}
connect();

function handle(msg) {
    if (msg.type === 'state') {
        gameState = msg;
        // Build color map from robots array order (stable)
        msg.robots.forEach((r, i) => { if (!playerColor[r.name]) playerColor[r.name] = COLORS[i % COLORS.length]; });
        updateHUD(); render();
    } else if (msg.type === 'auth_result') {
        showToast(msg.ok ? `Auth: ${msg.name}` : `Auth failed: ${msg.message}`);
    } else if (msg.type === 'admin_event') {
        if (msg.action === 'music') { toggleMusic(); setFeedback('♪'); return; }
        showToast(`Admin: ${msg.action.toUpperCase()}`);
        setFeedback(`✓ ${msg.action}`);
        if (msg.action === 'start' && _pendingAdminStart) { _pendingAdminStart = false; startRec(); }
        if (msg.action === 'reset') stopRec();
    } else if (msg.type === 'game_over') {
        showToast(`🏆 WINNER: ${msg.winner}`); stopRec();
    } else if (msg.type === 'error') {
        showToast(msg.message); setFeedback(`✗ ${msg.message}`);
    }
}

// ── HUD ───────────────────────────────────────────────────────────────────────
function updateHUD() {
    if (!gameState) return;
    const { running, paused, step, robots, prizes } = gameState;
    document.getElementById('ftick').textContent = `TICK: ${step}`;
    document.getElementById('fpri').textContent  = `PRIZES: ${prizes.length}`;
    let label = 'IDLE';
    if (running && !paused) { label = 'RUNNING'; setDot('run', 'RUNNING'); }
    else if (paused)         { label = 'PAUSED';  setDot('on', 'PAUSED'); }
    else                     { setDot('on', 'CONNECTED'); }
    document.getElementById('fstat').textContent = `STATUS: ${label}`;

    const html = robots.map((r, i) =>
        `<div class="sr"><span class="sn" style="color:${playerColor[r.name]||'#fff'}">${r.name}</span><span class="sv">${r.score}</span></div>`
    ).join('');
    document.getElementById('sl').innerHTML = html || '—';
}

// ── Render ────────────────────────────────────────────────────────────────────
function render() {
    ctx.clearRect(0, 0, W, H);
    ctx.fillStyle = '#0a0a12';
    ctx.fillRect(0, 0, W, H);
    if (!gameState) return;

    const cw = W / GRID_W, ch = H / GRID_H;

    // Grid lines
    ctx.strokeStyle = '#15152a';
    ctx.lineWidth = 0.4;
    for (let x = 0; x <= GRID_W; x++) { ctx.beginPath(); ctx.moveTo(x*cw,0); ctx.lineTo(x*cw,H); ctx.stroke(); }
    for (let y = 0; y <= GRID_H; y++) { ctx.beginPath(); ctx.moveTo(0,y*ch); ctx.lineTo(W,y*ch); ctx.stroke(); }

    // Obstacles
    ctx.fillStyle = '#1e1e3a';
    for (const [ox, oy] of gameState.obstacles) {
        ctx.fillRect(ox*cw+0.5, (GRID_H-1-oy)*ch+0.5, cw-1, ch-1);
    }

    // Prizes
    for (const pr of gameState.prizes) {
        const px = (pr.x + 0.5) * cw, py = (GRID_H - 1 - pr.y + 0.5) * ch;
        const r  = Math.min(cw, ch) * 0.38;
        const hue = 60 + (pr.value / 9) * 60; // yellow→green
        ctx.fillStyle = `hsl(${hue},85%,58%)`;
        ctx.beginPath(); ctx.arc(px, py, r, 0, Math.PI*2); ctx.fill();
        ctx.fillStyle = '#000';
        ctx.font = `bold ${Math.floor(r * 1.7)}px Courier New`;
        ctx.textAlign = 'center'; ctx.textBaseline = 'middle';
        ctx.fillText(pr.value, px, py);
    }

    // Robots
    for (const r of gameState.robots) {
        const rx = (r.x + 0.5) * cw, ry = (GRID_H - 1 - r.y + 0.5) * ch;
        const rr  = Math.min(cw, ch) * 0.44;
        const col = playerColor[r.name] || '#fff';
        // Glow
        ctx.shadowColor = col; ctx.shadowBlur = 10;
        ctx.fillStyle = col;
        ctx.beginPath(); ctx.arc(rx, ry, rr, 0, Math.PI*2); ctx.fill();
        ctx.shadowBlur = 0;
        ctx.fillStyle = '#000';
        ctx.font = `bold ${Math.floor(rr * 1.5)}px Courier New`;
        ctx.textAlign = 'center'; ctx.textBaseline = 'middle';
        ctx.fillText(r.name[0], rx, ry);
        // Name below robot
        ctx.fillStyle = col;
        ctx.font = `bold ${Math.floor(rr * 0.75)}px Courier New`;
        ctx.textBaseline = 'top';
        ctx.fillText(r.name, rx, ry + rr + 2);
    }
}

// ── Admin ──────────────────────────────────────────────────────────────────────
function adminAct(action) {
    if (!wsConnected) { showToast('Not connected'); return; }
    if (action === 'start') _pendingAdminStart = true;
    ws.send(JSON.stringify({ type:'admin', password: document.getElementById('ap').value, action }));
}
function setFeedback(t) {
    const el = document.getElementById('afb'); el.textContent = t;
    setTimeout(() => el.textContent = '', 3000);
}

// ── Splash ────────────────────────────────────────────────────────────────────
function dismissSplash() {
    document.getElementById('spl').style.display = 'none';
}

// ── Music ─────────────────────────────────────────────────────────────────────
function toggleMusic() {
    const bgm = document.getElementById('bgm');
    bgm.paused ? (bgm.currentTime=0, bgm.play().then(()=>updateMusicBtn(true)).catch(()=>{})) : (bgm.pause(), updateMusicBtn(false));
}
function updateMusicBtn(on) {
    const b = document.getElementById('mbtn'); b.textContent = on ? '♪ ON' : '♪ OFF';
    b.classList.toggle('play', on);
}

// ── Helpers ───────────────────────────────────────────────────────────────────
function setDot(cls, txt) {
    document.getElementById('dot').className = cls;
    document.getElementById('stxt').textContent = txt;
}
let _tt = null;
function showToast(t) {
    const el = document.getElementById('toast'); el.textContent = t; el.classList.add('show');
    clearTimeout(_tt); _tt = setTimeout(() => el.classList.remove('show'), 2800);
}

// Console helpers
// ── Video Recording ───────────────────────────────────────────────────────────
let _pendingAdminStart = false, _recorder = null, _recChunks = [];
function startRec() {
    if (_recorder && _recorder.state !== 'inactive') return;
    _recChunks = [];
    try {
        const stream = canvas.captureStream(30);
        const mime = ['video/webm;codecs=vp9','video/webm;codecs=vp8','video/webm']
            .find(m => MediaRecorder.isTypeSupported(m)) || 'video/webm';
        _recorder = new MediaRecorder(stream, {mimeType: mime});
        _recorder.ondataavailable = e => { if (e.data.size > 0) _recChunks.push(e.data); };
        _recorder.onstop = () => {
            const blob = new Blob(_recChunks, {type: _recorder.mimeType});
            const name = `ROBOWAR_${new Date().toISOString().replace(/[:.]/g,'-')}.webm`;
            const a = Object.assign(document.createElement('a'), {href: URL.createObjectURL(blob), download: name});
            a.click(); setTimeout(() => URL.revokeObjectURL(a.href), 5000);
            showToast('📹 Recording saved');
        };
        _recorder.start(1000);
        showToast('🔴 Recording started');
    } catch(e) { showToast('⚠ Recording not supported'); }
}
function stopRec() {
    if (_recorder && _recorder.state === 'recording') { _recorder.stop(); }
}

window.auth = (code) => ws && ws.readyState === 1 && ws.send(JSON.stringify({ type:'auth', code }));
window.move = (m)    => ws && ws.readyState === 1 && ws.send(JSON.stringify({ type:'cmd',  move: m }));
</script>
